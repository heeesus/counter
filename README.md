## Demo 
https://csslick.github.io/counter/

## Usage 
```
  counter_ani(class name)
```

## Example
```
    <section id="count">
      <div class="text-box">
        <h1>Counter</h1>
        <!-- data-num에 숫자값 설정 -->
        <p class="count" data-num="100">0</p>
        <p class="count" data-num="150">0</p>
        <p class="count" data-num="200">0</p>
      </div>
      
    </section>
    
    <!-- 문서 아래에 추가 -->
    <script src="counter.js"></script>
    
    <script>
      counter_ani(".count");
    </script>

```
